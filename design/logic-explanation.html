// ログイン処理の疑似コード

FUNCTION login(inputEmail, inputPassword)

  // ダミー入力値例
  // inputEmail = "user@example.com"
  // inputPassword = "password123"

  // 1. 入力値の空チェック
  IF inputEmail が空 THEN
    エラー「メールアドレスを入力してください」を追加
  ELSE
    email = inputEmail
  END IF

  IF inputPassword が空 THEN
    エラー「パスワードを入力してください」を追加
  ELSE
    password = inputPassword
  END IF

  IF エラーリストが空でない THEN
    RETURN ログイン画面（エラー表示）
  ELSE
    // 次の認証処理へ進む
  END IF

  // 2. ユーザー存在チェック
  user = userMapper.selectByEmail(email)

  IF user が null THEN
    エラー「メールアドレスまたはパスワードが正しくありません」を追加
    RETURN ログイン画面（エラー表示）
  ELSE
    // ユーザーが存在するためパスワード照合へ進む
  END IF

  // 3. パスワード照合
  IF BCrypt.matches(password, user.password) が false THEN
    エラー「メールアドレスまたはパスワードが正しくありません」を追加
    RETURN ログイン画面（エラー表示）
  ELSE
    // 認証成功
  END IF

  // 4. 認証成功時の処理
  authentication = UsernamePasswordAuthenticationToken(user, null, user.authorities)
  SecurityContext に authentication を保存
  RETURN リダイレクト("/tasks")

END FUNCTION

// タスク登録のバリデーション処理
// ダミー入力値例
// title    = ""
// dueDate  = "2026-01-01"
// priority = "1"
// status   = "TODO"

FUNCTION createTask(taskForm, bindingResult)

  // 1. @Valid による自動バリデーション
  IF bindingResult.hasErrors() THEN
    RETURN create.html（入力値を保持してフィールドエラー表示）
  ELSE
    // Service の業務ロジックバリデーションへ進む
  END IF

  // 2. Service での業務ロジックバリデーション
  // 期限の妥当性チェック
  IF taskForm.dueDate が過去の日付 THEN
    bindingResult.rejectValue("dueDate", "error.dueDate",
                              "期限は今日以降の日付を指定してください")
  ELSE
    // 期限は妥当
  END IF

  // タイトルの重複チェック（同一ユーザー内）
  existingTask = taskMapper.selectByTitleAndUserId(taskForm.title, userId)
  IF existingTask が存在する THEN
    bindingResult.rejectValue("title", "error.title",
                              "同じタイトルのタスクが既に存在します")
  ELSE
    // 同名タスクなし
  END IF

  // BindingResult に業務エラーが追加されたか確認
  IF bindingResult.hasErrors() THEN
    RETURN create.html（入力値を保持してエラー表示）
  ELSE
    // 登録処理へ進む
  END IF

  // 登録処理
  task = FormからEntityに変換(taskForm)
  task.userId = ログインユーザーID
  task.createdAt = 現在日時
  taskMapper.insert(task)

  RETURN リダイレクト("/tasks")（成功メッセージ付き）

END FUNCTION

// 複合検索のロジック（MyBatis XML形式）
// ダミー入力値例
// keyword    = "会議"
// categoryId = 2
// status     = "TODO"
// startDate  = "2026-05-01"
// endDate    = "2026-05-31"
// sortBy     = "dueDate"

<select id="search" resultType="Task">
  SELECT * FROM tasks
  WHERE user_id = #{userId}

  <!-- キーワード検索 -->
  <if test="keyword != null and keyword != ''">
    AND (title LIKE CONCAT('%', #{keyword}, '%')
         OR content LIKE CONCAT('%', #{keyword}, '%'))
  </if>

  <!-- カテゴリ絞り込み -->
  <if test="categoryId != null">
    AND category_id = #{categoryId}
  </if>

  <!-- ステータス絞り込み -->
  <if test="status != null and status != ''">
    AND status = #{status}
  </if>

  <!-- 期間絞り込み -->
  <if test="startDate != null">
    AND due_date >= #{startDate}
  </if>
  <if test="endDate != null">
    AND due_date <= #{endDate}
  </if>

  <!-- ソート -->
  ORDER BY
  <choose>
    <when test="sortBy == 'dueDate'">due_date</when>
    <when test="sortBy == 'priority'">priority DESC</when>
    <otherwise>created_at DESC</otherwise>
  </choose>
</select>

// データアクセス制御のロジック

FUNCTION updateTask(taskId, taskForm, loginUser)

  // タスクを取得
  task = taskMapper.selectById(taskId)

  // 存在チェック
  IF task が null THEN
    THROW NotFoundException("タスクが見つかりません")
  END IF

  // 権限チェック
  IF loginUser.role が "ADMIN" でない
     AND task.userId ≠ loginUser.id THEN
    THROW AccessDeniedException("このタスクを編集する権限がありません")
  END IF

  // 更新処理（権限OK）
  task = FormからEntityに変換(taskForm)
  task.updatedAt = 現在日時
  taskMapper.update(task)

  RETURN 成功

END FUNCTION

5 ステータス遷移のロジック

状態遷移図 必須
タスクのステータス変化の流れ
                    ┌─────────────────────────────────────────┐
                    │           タスクステータス遷移図          │
                    └─────────────────────────────────────────┘

                              ┌───────────┐
                              │   TODO    │ ← 新規作成
                              │  (未着手) │
                              └─────┬─────┘
                                    │
                                    │ 着手: 作業開始する
                                    ▼
                              ┌───────────┐
                      ┌────────│   DOING   │────────┐
                      │        │  (作業中) │        │
                      │        └─────┬─────┘        │
                      │              │              │
                      │ 保留: 一時停止 │ 完了: 作業完了 │ 中止: 作業中断
                      ▼              ▼              ▼
                ┌───────────┐  ┌───────────┐  ┌───────────┐
                │   HOLD    │  │   DONE    │  │ CANCELLED │
                │   (保留)  │  │   (完了)  │  │  (中止)   │
                └─────┬─────┘  └───────────┘  └───────────┘
                      │              ▲
                      │ 再開: 作業再開 │ 完了: 保留解除後に完了
                      └──────────────┘

          遷移可否:
            TODO      → DOING, CANCELLED
            DOING     → TODO, HOLD, DONE, CANCELLED
            HOLD      → DOING, DONE, CANCELLED
            DONE      → 遷移不可（最終状態）
            CANCELLED → 遷移不可（最終状態）
遷移可否の判定ロジック 必須
状態遷移が許可されるかの判定処理
// ステータス遷移の判定ロジック
// Enum 型を想定
// enum TaskStatus { TODO, DOING, HOLD, DONE, CANCELLED }
// ダミー入力値例
// currentStatus = TaskStatus.DOING
// newStatus     = TaskStatus.DONE

FUNCTION canTransition(currentStatus, newStatus)

  // 遷移ルールの定義
  allowedTransitions = {
    TaskStatus.TODO      → [TaskStatus.DOING, TaskStatus.CANCELLED],
    TaskStatus.DOING     → [TaskStatus.TODO, TaskStatus.HOLD, TaskStatus.DONE, TaskStatus.CANCELLED],
    TaskStatus.HOLD      → [TaskStatus.DOING, TaskStatus.DONE, TaskStatus.CANCELLED],
    TaskStatus.DONE      → [],  // 最終状態のため遷移不可
    TaskStatus.CANCELLED → []   // 最終状態のため遷移不可
  }

  // 現在のステータスから遷移可能なステータスを取得
  allowed = allowedTransitions[currentStatus]

  // 遷移可否を判定
  IF newStatus が allowed に含まれる THEN
    RETURN true
  ELSE
    RETURN false
  END IF

END FUNCTION


// ステータス変更処理

FUNCTION changeStatus(taskId, newStatus)

  task = taskMapper.selectById(taskId)

  IF task が null THEN
    THROW NotFoundException("タスクが見つかりません")
  END IF

  IF NOT canTransition(task.status, newStatus) THEN
    THROW IllegalStateException(
      "「" + task.status + "」から「" + newStatus + "」への遷移はできません"
    )
  ELSE
    // 遷移可能なためステータス更新へ進む
  END IF

  task.status = newStatus
  task.updatedAt = 現在日時

  // 完了時は完了日時を記録
  IF newStatus == TaskStatus.DONE THEN
    task.completedAt = 現在日時
  ELSE
    // 完了以外では completedAt を変更しない
  END IF

  taskMapper.updateStatus(task)

  RETURN task

END FUNCTION
